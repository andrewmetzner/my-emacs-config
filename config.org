  (defun java-eshell ()
    (interactive)
    (unless (buffer-file-name)
      (error "Buffer is not visiting a file"))
    (let* ((file (buffer-file-name))
           (classname (file-name-base file))
           (cmd (format "javac %s && java %s" (shell-quote-argument file) classname))
           (bufname (format "*Java Run: %s*" classname))
           eshell-buffer)
     
      (setq eshell-buffer (get-buffer bufname))
      (if eshell-buffer
      
          (switch-to-buffer-other-window eshell-buffer)
        
        (split-window-below)
        (other-window 1)
        (setq eshell-buffer (eshell 'N))
        (rename-buffer bufname)
        (switch-to-buffer eshell-buffer))
 
      (unless (get-buffer-process eshell-buffer)
        (eshell 'N))
      
      (let ((inhibit-read-only t))
        (erase-buffer))
   
      (goto-char (point-max))
      (insert cmd)
      (eshell-send-input)))

  (add-hook 'java-mode-hook
            (lambda ()
              (local-set-key (kbd "<f5>") #'java-eshell)))

    "Retrieve Jabber password for USER@SERVER from authinfo.gpg."
    (let ((found (auth-source-search
                  :host server
                  :user user
                  :require '(:user :secret)
                  :max 1)))

* overleaf style
#+BEGIN_SRC emacs-lisp

  (defvar my/org-exporting nil
    "Guard to prevent re-entrant PDF export triggered by save hooks.")

  (defun my/org-latex-export-and-view ()
    "Export org buffer to PDF via latexmk and refresh pdf-tools without losing focus."
    (interactive)
    (let ((my/org-exporting t)
          (current-window (selected-window))
          (pdf-path (concat (file-name-sans-extension (buffer-file-name)) ".pdf"))
          (log-path (concat (file-name-sans-extension (buffer-file-name)) ".log")))
      (org-latex-export-to-pdf)
      (if (file-exists-p pdf-path)
          (let ((pdf-buf (get-file-buffer pdf-path)))
            (if pdf-buf
                (with-current-buffer pdf-buf
                  (pdf-view-revert-buffer nil t))
              (find-file-noselect pdf-path))
            (select-window current-window)
            (message "PDF rebuilt at %s" (format-time-string "%H:%M:%S")))
        ;; PDF failed entirely — show the log
        (when (file-exists-p log-path)
          (find-file-other-window log-path)
          (goto-char (point-max))
          (re-search-backward "^!" nil t))
        (message "LaTeX export failed — see log"))))

  ;; Auto-export on save only when the PDF buffer is already open (Overleaf-style)
  (defun my/org-auto-export-on-save ()
    (when (and (not my/org-exporting)
               (eq major-mode 'org-mode)
               buffer-file-name
               (get-file-buffer (concat (file-name-sans-extension (buffer-file-name)) ".pdf")))
      (my/org-latex-export-and-view)))

  (add-hook 'after-save-hook #'my/org-auto-export-on-save)

#+END_SRC

  (defun my/org-git-push ()
    (interactive)
    (when (and buffer-file-name
               (string-prefix-p (expand-file-name "~/org/")
                                (file-name-directory buffer-file-name)))
      (let ((default-directory "~/org/"))
        (call-process "git" nil nil nil "add" ".")
        (call-process "git" nil nil nil "commit" "-m" "auto update")
        (call-process "git" nil nil nil "push"))))

  (add-hook 'after-save-hook #'my/org-git-push)
