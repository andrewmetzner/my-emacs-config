  (defun java-eshell ()
    (interactive)
    (unless (buffer-file-name)
      (error "Buffer is not visiting a file"))
    (let* ((file (buffer-file-name))
           (classname (file-name-base file))
           (cmd (format "javac %s && java %s" (shell-quote-argument file) classname))
           (bufname (format "*Java Run: %s*" classname))
           eshell-buffer)
     
      (setq eshell-buffer (get-buffer bufname))
      (if eshell-buffer
      
          (switch-to-buffer-other-window eshell-buffer)
        
        (split-window-below)
        (other-window 1)
        (setq eshell-buffer (eshell 'N))
        (rename-buffer bufname)
        (switch-to-buffer eshell-buffer))
 
      (unless (get-buffer-process eshell-buffer)
        (eshell 'N))
      
      (let ((inhibit-read-only t))
        (erase-buffer))
   
      (goto-char (point-max))
      (insert cmd)
      (eshell-send-input)))

  (add-hook 'java-mode-hook
            (lambda ()
              (local-set-key (kbd "<f5>") #'java-eshell)))

    (defun insert-system-time ()
      "Insert the current system time in ISO format with colon in timezone offset."
      (interactive)
      (let* ((raw (format-time-string "%Y-%m-%dT%H:%M:%S%z"))
             ;; Insert colon before last two digits of timezone
             (fixed (replace-regexp-in-string
                     "\\([+-][0-9][0-9]\\)\\([0-9][0-9]\\)$"
                     "\\1:\\2"
                     raw)))
        (insert fixed)))

  (defun my-jabber-get-password (server user)
    "Retrieve Jabber password for USER@SERVER from authinfo.gpg."
    (let ((found (auth-source-search
                  :host server
                  :user user
                  :require '(:user :secret)
                  :max 1)))
      (if found
          (let ((secret (plist-get (car found) :secret)))
            (if (functionp secret)
                (funcall secret)
              secret))
        (error "No credentials found for %s@%s" user server))))
