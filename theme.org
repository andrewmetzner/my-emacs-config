(defun my-anniversary-ymd (year month day)
  (let* ((now (decode-time (current-time)))
         (tyear year)
         (tmonth month)
         (tday day)
         (cyear (nth 5 now))
         (cmonth (nth 4 now))
         (cday (nth 3 now))
         (y (- cyear tyear))
         (m (- cmonth tmonth))
         (d (- cday tday)))

    (when (< d 0)
      (setq m (1- m))
      (setq d (+ d (calendar-last-day-of-month (if (= cmonth 1) 12 (1- cmonth))
                                               (if (= cmonth 1) (1- cyear) cyear)))))

    (when (< m 0)
      (setq y (1- y))
      (setq m (+ m 12)))
    (list y m d)))

(defun my-anniversary-ymd-string (year month day label)
  (let* ((ymd (my-anniversary-ymd year month day))
         (y (nth 0 ymd))
         (m (nth 1 ymd))
         (d (nth 2 ymd)))
    (format ";; %s - %d %s, %d %s, %d %s"
            label
            y (if (= y 1) "year" "years")
            m (if (= m 1) "month" "months")
            d (if (= d 1) "day" "days"))))
